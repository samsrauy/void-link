async function loadModule(id, path) { const target = document.getElementById(id); const GITHUB_BASE = "https://samsrauy.github.io/void-link/modules/"; // Force the path to be absolute to GitHub // If the path is just 'map/map.html', it becomes 'https://.../modules/map/map.html' const fullPath = path.startsWith('http') ? path : GITHUB_BASE + path; try { const response = await fetch(fullPath); if (!response.ok) throw new Error(`HTTP ${response.status}`); const html = await response.text(); target.innerHTML = html; // Execute scripts inside the module target.querySelectorAll('script').forEach(oldScript => { const newScript = document.createElement('script'); newScript.textContent = oldScript.textContent; document.body.appendChild(newScript); }); } catch (err) { console.error(`Injection Error (${fullPath}):`, err); target.innerHTML = `
LINK SEVERED: Check console for 404.
`; } }
