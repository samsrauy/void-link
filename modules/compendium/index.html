let datasworn = null; // Use the official RAW Github link for the latest Starforged schema const DATA_URL = "https://raw.githubusercontent.com/rsek/datasworn/master/datasworn/starforged/starforged.json"; async function init() { try { const res = await fetch(DATA_URL); if (!res.ok) throw new Error("Data Link Offline"); datasworn = await res.json(); populateMoves(); populateOracleCats(); // Truths are now under "setting_truths" renderTruths(); console.log("Compendium Sync Complete."); } catch (e) { console.error("Init Error:", e); document.getElementById('move-display').innerHTML = `
ERROR: DATABASE OFFLINE - ${e.message}
`; } } // --- MOVES LOGIC --- function populateMoves() { if(!datasworn) return; const moveCat = document.getElementById('move-cat'); moveCat.innerHTML = ""; // Datasworn Starforged moves are usually indexed by ID for (let key in datasworn.moves) { const category = datasworn.moves[key]; moveCat.add(new Option(category.name, key)); } loadMoves(); } function loadMoves() { const catKey = document.getElementById('move-cat').value; const list = document.getElementById('move-list'); list.innerHTML = ""; const contents = datasworn.moves[catKey].contents; for (let key in contents) { list.add(new Option(contents[key].name, key)); } displayMove(); } function displayMove() { const catKey = document.getElementById('move-cat').value; const moveKey = document.getElementById('move-list').value; const move = datasworn.moves[catKey].contents[moveKey]; // Fix: handle moves that use 'description' or 'text' const content = move.text || move.description || "No text available."; document.getElementById('move-display').innerHTML = `
${content.replace(/\n/g, '
')}
`; } // --- ORACLES LOGIC --- function populateOracleCats() { if(!datasworn) return; const oraCat = document.getElementById('ora-cat'); oraCat.innerHTML = ""; for (let key in datasworn.oracles) { oraCat.add(new Option(datasworn.oracles[key].name, key)); } loadOracleTables(); } function loadOracleTables() { const catKey = document.getElementById('ora-cat').value; const list = document.getElementById('ora-list'); list.innerHTML = ""; const cat = datasworn.oracles[catKey]; // Oracles can have 'contents' (tables) OR 'collections' (sub-folders) const tables = cat.contents || {}; for (let key in tables) { list.add(new Option(tables[key].name, key)); } // If there are sub-categories (like 'Space' -> 'Planets'), handle that here: if (cat.collections) { for (let subKey in cat.collections) { const subTables = cat.collections[subKey].contents; for (let tKey in subTables) { list.add(new Option(`${cat.collections[subKey].name}: ${subTables[tKey].name}`, `${subKey}/contents/${tKey}`)); } } } } function rollOracle() { if(!datasworn) return; const catKey = document.getElementById('ora-cat').value; const tablePath = document.getElementById('ora-list').value; let table; if (tablePath.includes('/')) { const parts = tablePath.split('/'); // handles sub-collections table = datasworn.oracles[catKey].collections[parts[0]].contents[parts[2]]; } else { table = datasworn.oracles[catKey].contents[tablePath]; } const roll = Math.floor(Math.random() * 100) + 1; const row = table.rows.find(r => roll >= r.floor && roll <= r.ceiling); document.getElementById('ora-display').innerHTML = `
${row.text || row.result}
Result: ${roll} | Table: ${table.name}
`; } // --- NPC GENERATOR --- function generateNPC() { // Starforged uses specific oracle IDs for characters const getName = () => { const table = datasworn.oracles['characters'].collections['name'].contents['given']; const r = Math.floor(Math.random() * 100) + 1; return table.rows.find(row => r >= row.floor && r <= row.ceiling).text; }; const getDetail = (sub, tab) => { const table = datasworn.oracles['characters'].contents[tab]; const r = Math.floor(Math.random() * 100) + 1; return table.rows.find(row => r >= row.floor && r <= row.ceiling).text; }; const name = getName(); const role = getDetail('characters', 'role'); const goal = getDetail('characters', 'goal'); const desc = getDetail('characters', 'descriptor'); document.getElementById('npc-display').innerHTML = `
${name}
${desc} ${role}
Goal: ${goal}
`; } function renderTruths() { if(!datasworn || !datasworn.setting_truths) return; const truths = datasworn.setting_truths; document.getElementById('truth-list').innerHTML = Object.keys(truths).map(k => `
${truths[k].name.toUpperCase()}
`).join(''); } // Initialization helper for the "INIT" tab function rollCreation(type) { if(!datasworn) return; let table; if (type === 'background') { table = datasworn.oracles['characters'].contents['background']; } else { table = datasworn.oracles['ship'].collections['name'].contents['given']; } const r = Math.floor(Math.random() * 100) + 1; const row = table.rows.find(row => r >= row.floor && row <= row.ceiling); if (type === 'background') document.getElementById('create-bg').innerText = row.text; else document.getElementById('create-ship').innerText = row.text; } init();
